```c

{
    /* Mismatch type interrupt */
    GPIO_INTERRUPT_ON_MISMATCH,
    /* Interrupt on rising edge */
    GPIO_INTERRUPT_ON_RISING_EDGE,
    /* Interrupt on falling edge */
    GPIO_INTERRUPT_ON_FALLING_EDGE,

    GPIO_INTERRUPT_ON_BOTH_EDGES,
```
